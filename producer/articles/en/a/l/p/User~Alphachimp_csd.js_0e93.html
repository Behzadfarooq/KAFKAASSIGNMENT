//URL parameter function:
//Originally from [[User:Lupin/autoedit.js]]
function gvo(paramName) {
  var cmdRe=RegExp('[&?]'+paramName+'=([^&]*)');
  var h=document.location;
  var m;
  if (m=cmdRe.exec(h)) {
    try { 
      return unescape(decodeURI(m[1]));
    } catch (someError) {}
  }
  return null;
};
