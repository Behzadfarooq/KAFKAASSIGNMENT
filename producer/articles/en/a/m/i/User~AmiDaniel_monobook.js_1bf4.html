function addOldids(){
    var re=/index[.]php[?]title=[^&]*&oldid=(\d+)$/;
    var len=document.links.length;
    for (var i=0; i<len; ++i) {
        var a=document.links[i];
        if (re.test(a.href)) {
            var m=a.href.match(re);
            a.innerHTML += ' (' + m[1] + ')';
        }
    }
}
