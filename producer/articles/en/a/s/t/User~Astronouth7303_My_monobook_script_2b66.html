/*

[edit] hooker


*/
function initLoginHooks() {
alert("Hook!");
if (o) {
  setHookByObj(o, "submit", checkFormSubmit);
  setHookByObj(o.wpName, "change", updateloginform);
  setHookByObj(o.wpPassword, "change", updateloginform);
  setHookByObj(o.wpRetype, "change", updateloginform);
  setHookByObj(o.wpEmail, "change", updateloginform);
  setHookByObj(o.wpRealName, "change", updateloginform);
}
}

/*

[edit] Submit callback


*/
function checkFormSubmit() {
  return checkpasswordmatch();
}

/*

[edit] Init


*/
var o=document.getElementById("userlogin"); 
var datacreate = true;
var datauserealname = true;
var datauseemail = true;

function initLogin() {
alert("Init");
o=document.getElementById("userlogin"); 
if (!o) {
  o = document.userlogin;
}
if (!o) {
  o = document.forms.item(0);
}
 datacreate = document.getElementById('wpRetype') ? true : false;
 datauserealname = document.getElementById('wpRealName') ? true : false;
 datauseemail = document.getElementById('wpEmail') ? true : false;
initLoginHooks();
updateloginform();

if (!o) {
  alert("(Init) Form not found!");
}
}

setHookByObj(window, "load", initLogin);

/*

[edit] update


*/
function updateloginform() {
        window.status = "Updating..."
        o=document.getElementById("userlogin");
        if (!o) {
                alert("(update) Form not found!");
        } else {
                if( datacreate ) {
                        o.wpCreateaccount.disabled=(o.wpRetype.value=="");
                }
                o.wpLoginattempt.disabled = (
                                      (
                                            // New user entries
                                        (
                                          datacreate && datauserealname && (
                                            o.wpRealName.value!=""
                                          )
                                        ) ||
                                            (
                                          datacreate && datauseemail && (
                                            o.wpEmail.value !=""
                                          )
                                        ) ||
                                        (
                                          o.wpRetype.value!=""
                                        )
                                      ) ||
                                      // Logins
                                          (
                                        o.wpPassword.value==""
                                      ) ||
                                          (
                                        o.wpName.value==""
                                      )
                                    );
        }
        window.status = "Updating...Done"
}
